60 // this is the ATCACommand parameter structure. The caller to the command method must
61 // initialize param1, param2 and data if appropriate. The command method will fill in the rest
62 // and initialize the packet so it's ready to send via the ATCAIFace.
63 // this particular structure mimics the ATSHA and ATECC family device's command structures
64 
65 // Note: pack @ 2 is required, @ 1 causes word alignment crash (though it should not), a known bug in GCC.
66 // @2, the wire still has the intended byte alignment with arm-eabi. this is likely the least portable part of atca
67 
68 
69 #ifdef ATCA_NO_PRAGMA_PACK
70 typedef struct __attribute__ ((packed))
71 #else
72 #pragma pack( push, ATCAPacket, 2 )
73 typedef struct
74 #endif
78 {
79 
80  // used for transmit/send
81  uint8_t _reserved; // used by HAL layer as needed (I/O tokens, Word address values)
82 
83  //--- start of packet i/o frame----
84  uint8_t txsize;
85  uint8_t opcode;
86  uint8_t param1; // often same as mode
87  uint16_t param2;
88  uint8_t data[192]; // includes 2-byte CRC. data size is determined by largest possible data section of any
89  // command + crc (see: x08 verify data1 + data2 + data3 + data4)
90  // this is an explicit design trade-off (space) resulting in simplicity in use
91  // and implementation
92  //--- end of packet i/o frame
93 
94  // used for receive
95  uint8_t execTime; // execution time of command by opcode
96 
97  // structure should be packed since it will be transmitted over the wire
98  // this method varies by compiler. As new compilers are supported, add their structure packing method here
99 
100 } ATCAPacket;
